\begin{frame}
  \frametitle{Introduction}
  \begin{itemize}
    \item  Model specification for:
    \begin{itemize}
      \item finite-state systems
      \item infinite-state systems
      \item synchronous systems 
    \end{itemize}
    \item Asynchronous systems not supported 
  \end{itemize}
\end{frame}

\begin{frame}
  \begin{itemize}
    \item Model checking algorithms:
      \begin{itemize}
        \item finite-state systems:
          \begin{itemize}
            \item BDD-based model checking
            \item SAT-based algorithms (BMC, K-Liveness, IC3)
          \end{itemize}
        \item infinite-state systems:
          \begin{itemize}
            \item SMT-based model checking 
          \end{itemize}
      \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Algorithms}
\begin{itemize}
  \item BDD-based algorithm
  \begin{itemize}
    \item LTL and CTL
    \item Heritated from $nuSMV$
    \item check\_ltlspec, check\_ctlspec
    
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Algorithms}
    \begin{itemize}
      \item SAT-based algorithms 
      \begin{itemize}
        \item LTL only
      \end{itemize}
    \item BMC
      \begin{itemize}
        \item Counter example finder in a time bound $k$
        \item check\_ltlspec\_bmc
      \end{itemize}
    \item IC3 and K-liveness
    \begin{itemize}
      \item Invariants checker
      \item Creates invariant inductively
      \item check\_invar\_ic3, check\_ltlspec\_ic3

    \end{itemize} 

    \end{itemize}

\end{frame}
